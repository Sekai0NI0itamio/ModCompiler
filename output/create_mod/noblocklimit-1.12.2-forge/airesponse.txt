We'll create a complete Forge 1.12.2 mod that removes the block limit for /fill and /clone commands, and optimizes them by processing in batches over multiple ticks to prevent server lag. The mod will also allow infinite block placement without throwing the "Too many blocks" error.

The implementation includes:
- Custom `CommandNoLimitFill` and `CommandNoLimitClone` that extend vanilla commands but remove the limit and add async batching.
- A `FillTask` class that stores fill parameters and processes them in chunks per world tick.
- A simple config for batch size.

Below are the required source files.

        return "fill";
    }

    @Override
    public String getUsage(ICommandSender sender) {
        return "/fill <from> <to> <block> [replace|destroy|outline|keep] [data] [<oldBlock> [<oldData>]]";
    }

    @Override
    public int getRequiredPermissionLevel() {
        return 2;
    }

    @Override
    public void execute(MinecraftServer server, ICommandSender sender, String[] args) throws CommandException {
        if (args.length < 6) throw new WrongUsageException(getUsage(sender));

        BlockPos from = parseBlockPos(sender, args, 0, false);
        BlockPos to = parseBlockPos(sender, args, 3, false);
        Block block = getBlockByText(sender, args[6]);

        int meta = 0;
        if (args.length >= 8) meta = parseInt(args[7], 0, 15);

        // Original FillMode enum
        FillMode mode = FillMode.REPLACE;
        if (args.length >= 9) {
            try {
                mode = FillMode.valueOf(args[8].toUpperCase());
            } catch (IllegalArgumentException e) {
                throw new CommandException("commands.fill.invalidMode", args[8]);
            }
        }

        IBlockState fillState = block.getStateFromMeta(meta);

        // Optional old block filtering
        Block oldBlock = null;
        int oldMeta = 0;
        if (args.length >= 10) {
            oldBlock = getBlockByText(sender, args[9]);
            if (args.length >= 11) oldMeta = parseInt(args[10], 0, 15);
        }

        // Apply filtering for modes
        switch (mode) {
            case DESTROY:
            case REPLACE:
            case KEEP:
            case OUTLINE:
                // No special handling needed for old block
                break;
            default:
                throw new CommandException("commands.fill.invalidMode", mode);
        }

        // Remove the limit check – allow infinite blocks
        // Create a FillTask to process in batches
        FillTask task = new FillTask(sender.getEntityWorld(), from, to, fillState, oldBlock, oldMeta, mode);
        FillTaskHandler.addTask(task);
        notifyCommandListener(sender, this, "commands.fill.start", from, to, task.getTotalBlocks());
    }

    @Override
    public List<String> getTabCompletions(MinecraftServer server, ICommandSender sender, String[] args, @Nullable BlockPos targetPos) {
        if (args.length < 3) return getTabCompletionCoordinate(args, 0, targetPos);
        if (args.length >= 6 && args.length < 9) return getTabCompletionCoordinate(args, 3, targetPos);
        return Collections.emptyList();
    }

    public enum FillMode {
        REPLACE, DESTROY, KEEP, OUTLINE
    }
}

package net.itamio.noblocklimit.task;

import net.itamio.noblocklimit.command.CommandNoLimitFill;
import net.itamio.noblocklimit.config.ModConfig;
import net.minecraft.block.Block;
import net.minecraft.block.state.IBlockState;
import net.minecraft.init.Blocks;
import net.minecraft.nbt.NBTTagCompound;
import net.minecraft.tileentity.TileEntity;
import net.minecraft.util.math.BlockPos;
import net.minecraft.world.World;

public class FillTask {
    private final World world;
    private final BlockPos from;
    private final BlockPos to;
    private final IBlockState state;
    private final Block oldBlock;
    private final int oldMeta;
    private final CommandNoLimitFill.FillMode mode;
    private final int totalBlocks;
    private int currentIndex = 0;
    private boolean done = false;

    public FillTask(World world, BlockPos from, BlockPos to, IBlockState state, Block oldBlock, int oldMeta, CommandNoLimitFill.FillMode mode) {
        this.world = world;
        this.from = from;
        this.to = to;
        this.state = state;
        this.oldBlock = oldBlock;
        this.oldMeta = oldMeta;
        this.mode = mode;
        int minX = Math.min(from.getX(), to.getX());
        int minY = Math.min(from.getY(), to.getY());
        int minZ = Math.min(from.getZ(), to.getZ());
        int maxX = Math.max(from.getX(), to.getX());
        int maxY = Math.max(from.getY(), to.getY());
        int maxZ = Math.max(from.getZ(), to.getZ());
        this.totalBlocks = (maxX - minX + 1) * (maxY - minY + 1) * (maxZ - minZ + 1);
    }

    public boolean isDone() { return done; }

    public int getTotalBlocks() { return totalBlocks; }

    public void processBatch() {
        if (done) return;
        int batchSize = ModConfig.batchSize;
        int processed = 0;

        int minX = Math.min(from.getX(), to.getX());
        int minY = Math.min(from.getY(), to.getY());
        int minZ = Math.min(from.getZ(), to.getZ());
        int maxX = Math.max(from.getX(), to.getX());
        int maxY = Math.max(from.getY(), to.getY());
        int maxZ = Math.max(from.getZ(), to.getZ());

        // Convert linear index to 3D coordinates
        while (currentIndex < totalBlocks && processed < batchSize) {
            int localIndex = currentIndex;
            int x = minX + (localIndex % (maxX - minX + 1));
            localIndex /= (maxX - minX + 1);
            int y = minY + (localIndex % (maxY - minY + 1));
            localIndex /= (maxY - minY + 1);
            int z = minZ + localIndex;

            BlockPos pos = new BlockPos(x, y, z);

            // Apply mode logic
            boolean shouldPlace = false;
            switch (mode) {
                case REPLACE:
                    shouldPlace = canReplace(pos);
                    break;
                case DESTROY:
                    world.destroyBlock(pos, true);
                    shouldPlace = true;
                    break;
                case KEEP:
                    shouldPlace = world.isAirBlock(pos);
                    break;
                case OUTLINE:
                    // Only edges
                    if (x == minX || x == maxX || y == minY || y == maxY || z == minZ || z == maxZ) {
                        shouldPlace = canReplace(pos);
                    }
                    break;
                default:
                    shouldPlace = canReplace(pos);
            }

            if (shouldPlace) {
                world.setBlockState(pos, state, 3);
                // Preserve tile entity?
                // We could also copy tile entity data if needed
            }

            currentIndex++;
            processed++;
        }

        if (currentIndex >= totalBlocks) {
            done = true;
        }
    }

    private boolean canReplace(BlockPos pos) {
        IBlockState oldState = world.getBlockState(pos);
        if (oldBlock == null) {
            return true;
        }
        Block b = oldState.getBlock();
        int m = b.getMetaFromState(oldState);
        return b == oldBlock && (oldMeta == -1 || m == oldMeta);
    }
}

package net.itamio.noblocklimit.task;

import net.minecraftforge.fml.common.eventhandler.SubscribeEvent;
import net.minecraftforge.fml.common.gameevent.TickEvent;

import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;

public class FillTaskHandler {
    private static final List<FillTask> tasks = new ArrayList<>();
    private static final List<FillTask> toAdd = new ArrayList<>();

    public static void addTask(FillTask task) {
        toAdd.add(task);
    }

    @SubscribeEvent
    public void onWorldTick(TickEvent.WorldTickEvent event) {
        if (event.phase != TickEvent.Phase.END) return;
        // Add new tasks
        synchronized (tasks) {
            tasks.addAll(toAdd);
            toAdd.clear();
        }
        // Process one batch per tick per task (or limit)
        // For simplicity, process each task one batch per tick
        Iterator<FillTask> it = tasks.iterator();
        final int maxTasksPerTick = 5; // avoid too many
        int count = 0;
        while (it.hasNext() && count < maxTasksPerTick) {
            FillTask task = it.next();
            task.processBatch();
            if (task.isDone()) {
                it.remove();
            }
            count++;
        }
    }
}
